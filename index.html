function parseTSDB(events){ return events.map(e=>{ const hasScore=e.intHomeScore!==null&&e.intHomeScore!==''&&e.intHomeScore!==undefined; const isLive=hasScore&&e.strStatus&&(e.strStatus.includes("'")||['HT','1H','2H','ET','PEN'].includes(e.strStatus)); return{ _id:e.idEvent, homeTeam:{name:e.strHomeTeam,crest:e.strHomeTeamBadge||null}, awayTeam:{name:e.strAwayTeam,crest:e.strAwayTeamBadge||null}, competition:{name:e.strLeague,code:null}, utcDate:e.dateEvent+'T'+(e.strTime||'12:00:00')+'Z', status:isLive?'IN_PLAY':hasScore?'FINISHED':'TIMED', venue:e.strVenue||null, minute:isLive?e.strStatus:null, score:{fullTime:{ home:hasScore?parseInt(e.intHomeScore):null, away:hasScore?parseInt(e.intAwayScore):null, }} }; }); } async function fetchTSDB(){ const d=todayS(); const urls=[ `${TSDB_BASE}/eventsday.php?d=${d}&s=Soccer`, `https://corsproxy.io/?url=${encodeURIComponent('https://www.thesportsdb.com/api/v1/json/3/eventsday.php?d='+d+'&s=Soccer')}`, `https://api.allorigins.win/raw?url=${encodeURIComponent('https://www.thesportsdb.com/api/v1/json/3/eventsday.php?d='+d+'&s=Soccer')}`, ]; for(const url of urls){ try{ const r=await fetch(url,{signal:AbortSignal.timeout(8000)}); if(!r.ok)continue; const j=await r.json(); const events=j.events||[]; if(events.length>0) return parseTSDB(events); }catch(e){continue;} } throw new Error('Tous les proxys ont échoué'); }
